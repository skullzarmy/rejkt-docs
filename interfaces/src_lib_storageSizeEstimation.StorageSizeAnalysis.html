    Interface StorageSizeAnalysis

    Complete storage size analysis result

    This interface represents the comprehensive analysis of an IndexedDB object store, providing aggregated metrics, size distributions, and status indicators to help with storage optimization decisions.

    interface StorageSizeAnalysis {
        averageItemSize: number;
        breakdown: { [key: string]: StorageSizeBreakdown };
        browserStorageLimit: number;
        itemCount: number;
        largestItem: { key: string; size: number };
        smallestItem: { key: string; size: number };
        storageLimitExceeded: boolean;
        storagePercentageUsed: number;
        totalSize: number;
    }
